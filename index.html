/** * KEMASKINI KOD INI DI GOOGLE APPS SCRIPT EDITOR */ function doGet(e) { const action = e.parameter.action; const sheetId = "11dkbToz73vc1PZwF1LT-epIMnns_1Ae1HDfeISzViZU"; const ss = SpreadsheetApp.openById(sheetId); const sheet = ss.getSheets()[0]; // Ambil semua data dari sheet const data = sheet.getDataRange().getValues(); const headers = data[0]; const allRows = data.slice(1).filter(row => row[1] !== ""); // Filter baris yang mempunyai nama if (action === 'read') { let rows = allRows; // Logik Filter mengikut kelas const kelasFilter = e.parameter.kelas; if (kelasFilter && kelasFilter !== "Semua") { // Indeks 4 merujuk kepada kolum 'KELAS' rows = rows.filter(row => row[4] == kelasFilter); } return ContentService.createTextOutput(JSON.stringify({ rows: rows })) .setMimeType(ContentService.MimeType.JSON); } // Fungsi tambahan untuk mendapatkan senarai kelas unik bagi menu filter if (action === 'getClasses') { const classes = allRows.map(row => row[4]); const uniqueClasses = [...new Set(classes)].filter(c => c !== "").sort(); return ContentService.createTextOutput(JSON.stringify({ classes: uniqueClasses })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { const sheetId = "11dkbToz73vc1PZwF1LT-epIMnns_1Ae1HDfeISzViZU"; const ss = SpreadsheetApp.openById(sheetId); const sheet = ss.getSheets()[0]; const postData = JSON.parse(e.postData.contents); if (postData.action === 'update') { const rowIndex = postData.index; const marks = postData.marks; // Kemaskini kolum HP1, HP5, T2-KUMP, T2-INDT // Pastikan indeks kolum adalah betul (F=6, G=7, I=9, J=10) sheet.getRange(rowIndex + 1, 6).setValue(marks.hp1); sheet.getRange(rowIndex + 1, 7).setValue(marks.hp5); sheet.getRange(rowIndex + 1, 9).setValue(marks.t2kump); sheet.getRange(rowIndex + 1, 10).setValue(marks.t2indt); return ContentService.createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } }
